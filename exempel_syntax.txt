	skapa metoden <test_metod> <param1>, <param2>
	start
		skapa <summa> = <param1> + <param2>

	slut

	skapa metoden <test_metod2>

		skriv "Start Nu!"
	slut


	# Exempel på variabler

	skapa <var1>
	<var1> = 5

	skapa <var2> = 10
	skapa <var3> = <var1>

	skapa <var4> = 20
	<var4> = "Hej värld!"

	# Conditionals

	om 5 < 10
	start
		skriv "Det är större"
	slut

	om 5 < 10
	start
		skriv "Det är större"
	annars
		skriv "Det är mindre"
	slut

	# Iteratorer
	
	medans <var1> < 10
	start
		skriv <var1> + " är mindre än 10"
		<var1> = <var1> + 1      ELLER <var1> += 1
	slut

	för <i> 5 till 10
	start
		skriv i
	slut



	# Språket ska ha stöd för två typer av behållare,
	# dynamisk lista och hashtabell (parlista)

	skapa <min_lista> = Lista 1,2,3,4,5
	lägg till 6 <min_lista>
	för varje nummer i <min_lista>
	start
		skriv nummer
	slut
	skriv "Första numret i listan är: "  + <min_lista>[0]

	skapa <min_parlista> = ParLista "one":1,"two":2,"three":3
	<min_parlista>["four"] = 4
	lägg till 5 <min_parlista>["five"]
	för varje nyckel,värde i <min_parlista>
	start
		skriv nyckel + " " + värde
	slut
	skriv "Värdet för nyckel 'one' är: " + <min_parlista>["one"]
